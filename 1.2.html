JavaScript Concepts Example

String Methods

      const str = "Hello, World!";
      console.log("String Length:", str.length);
      console.log("Uppercase:", str.toUpperCase());
      console.log("Substring:", str.substring(0, 5));
      console.log("Index of 'World':", str.indexOf("World"));
      console.log("Replace 'Hello' with 'Hi':", str.replace("Hello", "Hi"));

Number Methods

      const num = 42.567;
      console.log("Fixed to 2 decimal places:", num.toFixed(2));
      console.log("Converted to exponential notation:", num.toExponential());
      console.log("Rounded:", Math.round(num));
      console.log("Square root:", Math.sqrt(num));
      console.log("Maximum of 10 and 20:", Math.max(10, 20));

Array Methods

      const mixedArray = ['apple', 5, 'banana', 10, 'cherry'];
      console.log("Array length:", mixedArray.length);
      console.log("Index of 'banana':", mixedArray.indexOf('banana'));
      console.log("Last index of number 10:", mixedArray.lastIndexOf(10));
      console.log("Array joined as string:", mixedArray.join(', '));
      console.log("Reversed array:", mixedArray.reverse());

Date Methods

      const currentDate = new Date();
      console.log("Current date:", currentDate.toDateString());
      console.log("Current time:", currentDate.toLocaleTimeString());
      console.log("Day of the week:", currentDate.getDay());
      console.log("Month:", currentDate.getMonth());
      console.log("Year:", currentDate.getFullYear());

Function Methods

      function greet(name) {
        return "Hello, " + name + "!";
      }
      const greetFunction = greet;
      console.log(greetFunction("Alice"));

      const multiply = function(x, y) {
        return x * y;
      };
      console.log("Result of multiplication:", multiply(5, 3));
